\newcommand{\newsec}[1]{\vspace{-0.5ex}\textbf{#1}\\}

% Used to place a label above an equals sign to easily
% reference and explain steps in derivations.
\newcommand{\annotate}[2]{\stackrel{\mathrm{#1}}{#2}}

% For \hfill-like functionality in math environments
\makeatletter
\newcommand{\pushright}[1]{\ifmeasuring@\hfill\text{#1}\else\omit$\displaystyle\hfill\text{#1}$\ignorespaces\fi}
\makeatother

% for figures in multicols environment
\newenvironment{multicolFigure}
  {\par\smallskip\noindent
  \minipage{\columnwidth}}
  {\endminipage\par\smallskip}

% center environment with reduced top and bottom spacing
% used to hold figures
\newenvironment{tightcenter}[1][\topsep]
  {\setlength{\topsep}{#1}\par\kern2pt\centering}% \begin{tightcenter}
  {\par\kern0pt}% \end{tightcenter}

% redefine \tableofcontents to not print "Contents"
\makeatletter
\renewcommand{\tableofcontents}[1][\contentsname]{%
  \section*{#1}
  \begin{multicols}{2}
    \@starttoc{toc}
  \end{multicols}
}
\makeatother

% Define align environments that display math as \textstyle
% See https://tex.stackexchange.com/a/274456
\newenvironment{talign}
 {\let\displaystyle\textstyle\align}
 {\endalign}
\newenvironment{talign*}
 {\let\displaystyle\textstyle\csname align*\endcsname}
 {\endalign}
